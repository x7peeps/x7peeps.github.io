MSSQL xp_cmdshell/OPENROWSET/BULK INSERT/Linked Server RCEдёҺжЁӘеҗ‘з§»еҠЁжҸҗжқғеҲ©з”ЁжҠҖжңҜ

MSSQL xp_cmdshell/OPENROWSET/BULK INSERT/Linked Server RCEдёҺжЁӘеҗ‘з§»еҠЁжҸҗжқғеҲ©з”ЁжҠҖжңҜ

Microsoft SQL Server жҳҜдјҒдёҡзҺҜеўғдёӯжңҖе№ҝжіӣйғЁзҪІзҡ„е…ізі»ж•°жҚ®еә“д№ӢдёҖпјҢдёҺ Windows з”ҹжҖҒе’Ң Active Directory ж·ұеәҰйӣҶжҲҗгҖӮдёҖдёӘе…ёеһӢз”ҹдә§йғЁзҪІйҮҢпјҢMSSQL иҮіе°‘еҗҢж—¶жҡҙйңІдәҶд»ҘдёӢж”»еҮ»йқўпјҡ

  • TDS Protocol йқўпјҡTabular Data Stream еҚҸи®®з«ҜеҸЈпјҲй»ҳи®Ө 1433пјҢTLS дёә 1434пјҢNamed Pipe дҪҝз”Ё SMBпјүпјҢеӨ„зҗҶжүҖжңү SQL жҹҘиҜў
  • xp_cmdshell йқўпјҡжү©еұ•еӯҳеӮЁиҝҮзЁӢеҸҜзӣҙжҺҘжү§иЎҢж“ҚдҪңзі»з»ҹе‘Ҫд»ӨпјҢе®һзҺ° RCE
  • OPENROWSET йқўпјҡеҸҜи®ҝй—®иҝңзЁӢж•°жҚ®жәҗгҖҒжү§иЎҢж–Үд»¶ж“ҚдҪңгҖҒе»әз«Ӣ Ad Hoc иҝһжҺҘ
  • BULK INSERT йқўпјҡеҸҜд»Һж–Үд»¶зі»з»ҹжү№йҮҸеҜје…Ҙж•°жҚ®еҲ°ж•°жҚ®еә“иЎЁ
  • Linked Server йқўпјҡеҸҜе»әз«ӢеҲ°е…¶д»– SQL Server жҲ– OLE DB ж•°жҚ®жәҗзҡ„й“ҫжҺҘпјҢж”ҜжҢҒеҲҶеёғејҸжҹҘиҜў
  • OLE Automation йқўпјҡsp_OACreate/sp_OAMethod зӯүеҸҜи°ғз”Ё COM еҜ№иұЎ
  • CLR Integration йқўпјҡCommon Language Runtime еҸҜеҠ иҪҪ .NET зЁӢеәҸйӣҶ
  • жқғйҷҗжҸҗеҚҮйқўпјҡdb_owner и§’иүІгҖҒdbo жқғйҷҗгҖҒsa иҙҰжҲ·гҖҒWindows и®ӨиҜҒйӣҶжҲҗ

еҜ№ж”»еҮ»иҖ…жқҘиҜҙпјҢMSSQL зҡ„д»·еҖјдёҚеңЁдәҺжҹҗдёӘеҚ•дёҖжјҸжҙһпјҢиҖҢеңЁдәҺе®ғжҠҠж•°жҚ®еӯҳеӮЁгҖҒж–Үд»¶ж“ҚдҪңгҖҒе‘Ҫд»Өжү§иЎҢгҖҒеҹҹи®ӨиҜҒдёҺеҲҶеёғејҸжҹҘиҜўиғҪеҠӣйӣҶдёӯеңЁеҗҢдёҖиҝӣзЁӢйҮҢгҖӮдёҖж—ҰиҺ·еҫ—ж•°жҚ®еә“и®ҝй—®жқғйҷҗпјҲејұеҜҶз ҒгҖҒй»ҳи®ӨеҮӯжҚ®гҖҒSQL жіЁе…ҘпјүпјҢж”»еҮ»иҖ…еҸҜд»ҘйҖҡиҝҮ xp_cmdshell жү§иЎҢзі»з»ҹе‘Ҫд»ӨгҖҒйҖҡиҝҮ OPENROWSET иҜ»еҸ–д»»ж„Ҹж–Үд»¶жҲ–еӨ–еёҰж•°жҚ®гҖҒйҖҡиҝҮ BULK INSERT еҜје…Ҙж–Үд»¶еҶ…е®№гҖҒйҖҡиҝҮ Linked Server жЁӘеҗ‘з§»еҠЁеҲ°е…¶д»–ж•°жҚ®еә“е®һдҫӢпјҢз”ҡиҮіеҲ©з”Ё Windows и®ӨиҜҒжңәеҲ¶зӣҙжҺҘи®ҝй—®еҹҹиө„жәҗгҖӮ

-- еҪ“еүҚз”ЁжҲ·
SELECT SYSTEM_USER AS CurrentUser, USER AS DatabaseUser;

-- еҪ“еүҚзҷ»еҪ•дҝЎжҒҜ

    SUSER_NAME() AS LoginName,
    ORIGINAL_LOGIN() AS OriginalLogin;

-- жҳҜеҗҰдёә sysadmin
SELECT IS_SRVROLEMEMBER('sysadmin') AS IsSysAdmin;

-- жңҚеҠЎеҷЁи§’иүІжҲҗе‘ҳ
SELECT 
    sp.name AS LoginName,
    sp.type_desc AS LoginType,
    sp.is_disabled AS Disabled,
    sp.default_database_name AS DefaultDB
FROM sys.server_principals sp
WHERE sp.type IN ('S', 'U', 'G')
ORDER BY sp.name;

-- ж•°жҚ®еә“и§’иүІжҲҗе‘ҳ
SELECT 
    dp.name AS UserName,
    dp.type_desc AS UserType,
    r.name AS RoleName
FROM sys.database_principals dp
LEFT JOIN sys.database_role_members drm ON dp.principal_id = drm.member_principal_id
LEFT JOIN sys.database_principals r ON drm.role_principal_id = r.principal_id
ORDER BY dp.name;

2.2 ж•°жҚ®еә“жһҡдёҫ

-- жүҖжңүж•°жҚ®еә“

SELECT 
    TABLE_SCHEMA,
    TABLE_NAME,
    TABLE_TYPE

-- жүҖжңүиЎЁпјҲеҢ…жӢ¬зі»з»ҹиЎЁпјү

    s.name AS SchemaName,
    t.name AS TableName,
    t.type_desc,
    p.rows AS RowCount
FROM sys.tables t
INNER JOIN sys.schemas s ON t.schema_id = s.schema_id
INNER JOIN sys.partitions p ON t.object_id = p.object_id AND p.index_id IN (0, 1)
ORDER BY s.name, t.name;
