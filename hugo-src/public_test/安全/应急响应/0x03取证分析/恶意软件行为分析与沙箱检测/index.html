  • дҝқеӯҳдёә RAW ж јејҸд»Ҙдҝқз•ҷе®Ңж•ҙдҝЎжҒҜ

2. зҪ‘з»ңжөҒйҮҸж·ұеәҰеҲҶжһҗ

зҪ‘з»ңжөҒйҮҸеҲҶжһҗжҳҜжҒ¶ж„ҸиҪҜд»¶иЎҢдёәеҲҶжһҗзҡ„йҮҚиҰҒз»„жҲҗйғЁеҲҶпјҢиғҪеӨҹжҸӯзӨә C2 йҖҡдҝЎгҖҒж•°жҚ®еӨ–жі„е’ҢжЁӘеҗ‘з§»еҠЁзӯүе…ій”®иЎҢдёәгҖӮ

PCAP еҲҶжһҗ

# дҪҝз”Ё TShark жҸҗеҸ– DNS жҹҘиҜў
tshark -r capture.pcap -Y "dns.qry.type == 1" -T fields -e dns.qry.name -e dns.a

# дҪҝз”Ё TShark жҸҗеҸ– HTTP иҜ·жұӮ
tshark -r capture.pcap -Y "http.request" -T fields -e ip.src -e ip.dst -e http.host -e http.request.uri -e http.request.method

# дҪҝз”Ё TShark жҸҗеҸ– TLS жҸЎжүӢдёӯзҡ„ SNI
tshark -r capture.pcap -Y "tls.handshake.type == 1" -T fields -e tls.handshake.extensions_server_name

# дҪҝз”Ё NetworkMiner жҸҗеҸ–ж–Үд»¶е’ҢзҪ‘з»ңе·Ҙд»¶
# NetworkMiner жҸҗдҫӣ GUI з•ҢйқўпјҢеҸҜиҮӘеҠЁи§Јжһҗ PCAP дёӯзҡ„ж–Үд»¶гҖҒеӣҫзүҮгҖҒиҜҒд№Ұзӯү

еҸ–иҜҒиҰҒзӮ№пјҡ

  • DNS жҹҘиҜўи®°еҪ•еҸҜжҸӯзӨә C2 еҹҹеҗҚе’Ң DGAпјҲеҹҹеҗҚз”ҹжҲҗз®—жі•пјү
  • HTTP User-Agent еӯ—з¬ҰдёІеҸҜиҜҶеҲ«жҒ¶ж„ҸиҪҜд»¶е®¶ж—Ҹ
  • TLS иҜҒд№ҰжҢҮзә№пјҲJA3/JA3SпјүеҸҜз”ЁдәҺе…іиҒ”е·ІзҹҘжҒ¶ж„ҸеҹәзЎҖи®ҫж–Ҫ
  • жҸҗеҸ–зҡ„еҸҜжү§иЎҢж–Үд»¶е’Ңж–ҮжЎЈйңҖиҝӣдёҖжӯҘиҝӣиЎҢе“ҲеёҢжҜ”еҜ№е’ҢжІҷз®ұеҲҶжһҗ

C2 йҖҡдҝЎжЈҖжөӢ

жҒ¶ж„ҸиҪҜд»¶зҡ„ C2 йҖҡдҝЎйҖҡеёёе…·жңүд»ҘдёӢзү№еҫҒпјҡ

  • е‘ЁжңҹжҖ§ HTTP/HTTPS иҜ·жұӮпјҡBeacon жҢүеӣәе®ҡй—ҙйҡ”еҗ‘ C2 еҸ‘йҖҒеҝғи·іиҜ·жұӮпјҢйҖҡиҝҮиҜ·жұӮй—ҙйҡ”зҡ„з»ҹи®ЎеҲҶжһҗеҸҜиҜҶеҲ«ејӮеёёе‘ЁжңҹжҖ§жөҒйҮҸ
  • DNS йҡ§йҒ“пјҡж”»еҮ»иҖ…еҲ©з”Ё DNS жҹҘиҜўдј иҫ“ж•°жҚ®пјҢзү№еҫҒеҢ…жӢ¬еӨ§йҮҸ TXT и®°еҪ•жҹҘиҜўгҖҒејӮеёёй•ҝзҡ„еӯҗеҹҹеҗҚгҖҒй«ҳйў‘ DNS иҜ·жұӮ
  • еҹҹеүҚзҪ®пјҲDomain FrontingпјүпјҡеҲ©з”Ё CDN йҡҗи—Ҹзңҹе®һ C2 жңҚеҠЎеҷЁпјҢйҖҡиҝҮжҜ”иҫғ TLS SNI е’Ң HTTP Host еӨҙеҸҜжЈҖжөӢ
# жЈҖжөӢе‘ЁжңҹжҖ§ HTTP иҜ·жұӮ
tshark -r capture.pcap -Y "http.request" -T fields -e frame.time -e ip.dst -e http.host | sort

# жЈҖжөӢ DNS йҡ§йҒ“зү№еҫҒ
tshark -r capture.pcap -Y "dns" -T fields -e dns.qry.name -e dns.qry.type | awk '{print length($1), $1}' | sort -rn | head -20

# з»ҹи®Ў DNS жҹҘиҜўйў‘зҺҮ
tshark -r capture.pcap -Y "dns.qry.type == 1" -T fields -e dns.qry.name | sort | uniq -c | sort -rn | head -20

ж•°жҚ®еӨ–жі„жЈҖжөӢ

ж•°жҚ®еӨ–жі„жҳҜжҒ¶ж„ҸиҪҜд»¶зҡ„жңҖз»Ҳзӣ®ж Үд№ӢдёҖпјҢеёёи§Ғж–№ејҸеҢ…жӢ¬пјҡ

  • еӨ§йҮҸ HTTP POST иҜ·жұӮеҗ‘еӨ–йғЁжңҚеҠЎеҷЁдёҠдј ж•°жҚ®
  • йҖҡиҝҮ DNS жҹҘиҜўзј–з ҒеӨ–дј ж•Ҹж„ҹж•°жҚ®
  • йҖҡиҝҮеҠ еҜҶйҖҡйҒ“пјҲTLSпјүдј иҫ“еӨ§йҮҸж•°жҚ®
# з»ҹи®Ў HTTP POST иҜ·жұӮзҡ„ж•°жҚ®йҮҸ
tshark -r capture.pcap -Y "http.request.method == POST" -T fields -e ip.dst -e http.host -e http.content_length | sort

# жЈҖжөӢејӮеёёеӨ§йҮҸзҡ„еҮәз«ҷж•°жҚ®дј иҫ“
tshark -r capture.pcap -Y "tcp.dstport == 443" -T fields -e ip.dst -e tcp.len | awk '{sum[$1]+=$2} END {for (ip in sum) print sum[ip], ip}' | sort -rn

4. YARA и§„еҲҷзј–еҶҷ

YARA жҳҜжҒ¶ж„ҸиҪҜд»¶жЈҖжөӢе’ҢеҲҶзұ»зҡ„з‘һеЈ«еҶӣеҲҖпјҢйҖҡиҝҮзј–еҶҷи§„еҲҷеҢ№й…ҚжҒ¶ж„ҸиҪҜд»¶зҡ„зү№еҫҒеӯ—з¬ҰдёІгҖҒеӯ—иҠӮеәҸеҲ—е’ҢиЎҢдёәжЁЎејҸгҖӮ

YARA и§„еҲҷзӨәдҫӢ

rule Emotet_Loader {
    meta:
        description = "Detects Emotet loader based on strings and byte patterns"
        author = "Forensic Analyst"
        date = "2026-06-24"
        hash = "example_sha256_hash"
        reference = "Emotet Campaign Analysis"

    strings:
        $s1 = "RegOpenKeyExA" ascii
        $s2 = "CryptEncrypt" ascii
        $s3 = "InternetOpenA" ascii
        $s4 = "CreateToolhelp32Snapshot" ascii
        $hex1 = { 8B 45 ?? 83 C0 ?? 8B 4D ?? E8 ?? ?? ?? ?? }
        $hex2 = { 68 ?? ?? ?? ?? 6A 00 6A 00 FF 55 ?? 89 45 ?? }

    condition:
        uint16(0) == 0x5A4D and
        filesize < 5MB and
        3 of ($s*) and
        1 of ($hex*)
}

YARA жү«жҸҸе‘Ҫд»Ө

# дҪҝз”Ё YARA жү«жҸҸеҚ•дёӘж–Үд»¶
yara emotet_rules.yar malware.exe

# йҖ’еҪ’жү«жҸҸж•ҙдёӘзӣ®еҪ•
yara -r emotet_rules.yar /path/to/suspicious/files/

# жү«жҸҸе№¶жҳҫзӨәеҢ№й…Қзҡ„и§„еҲҷеҗҚз§°
yara -r -s emotet_rules.yar /path/to/files/

# дҪҝз”ЁеӨҡзәҝзЁӢеҠ йҖҹжү«жҸҸ
yara -r -p 4 emotet_rules.yar /path/to/files/

# жү«жҸҸеҶ…еӯҳдёӯзҡ„иҝӣзЁӢ
yara -r emotet_rules.yar /proc/PID/mem

еҸ–иҜҒиҰҒзӮ№пјҡ

  • и§„еҲҷеә”еҢ…еҗ«и¶іеӨҹзҡ„зү№ејӮжҖ§жқЎд»¶пјҢйҒҝе…ҚиҜҜжҠҘ
  • дҪҝз”Ё uint16(0) == 0x5A4D йҷҗе®ҡ PE ж–Үд»¶зұ»еһӢ
  • з»“еҗҲеӯ—з¬ҰдёІеҢ№й…Қе’ҢеҚҒе…ӯиҝӣеҲ¶еӯ—иҠӮеәҸеҲ—жҸҗй«ҳжЈҖжөӢзІҫеәҰ
  • е®ҡжңҹжӣҙж–°и§„еҲҷд»ҘиҰҶзӣ–ж–°зҡ„жҒ¶ж„ҸиҪҜд»¶еҸҳз§Қ

5. еЁҒиғҒжғ…жҠҘйӣҶжҲҗ

е°ҶжҒ¶ж„ҸиҪҜд»¶еҲҶжһҗз»“жһңдёҺеЁҒиғҒжғ…жҠҘе№іеҸ°йӣҶжҲҗпјҢеҸҜд»Ҙдё°еҜҢ IOC дёҠдёӢж–ҮгҖҒе…іиҒ”ж”»еҮ»жҙ»еҠЁе№¶иҝҪиёӘеЁҒиғҒиЎҢдёәиҖ…гҖӮ
